        """
        Replace multiples and units strings by numbers
        """

        equation = re.compile("[0-9]")
        print(query_string)

        if equation.search(query_string):
            multiple = re.compile('|'.join(aliases.multiples.keys()))
            result = multiple.sub(lambda x: aliases.multiples[x.group()], query_string)
            unit = re.compile('[A-Za-z]+$')
            result = unit.sub('', result).replace(" ", "")
            print(result)

        return



[docs]    def prepare_query(self, query_string):
        """
        Split a query string into keywords and interpretate boolean operators
            :parameter query_string: unformatted string

        Output :
            :return OR: list of terms to search alternatively
            :return AND: list of sets to search together

        """

        OR_split = query_string.split('OR')

        self.AND = []
        self.OR = []

        for element in OR_split:
            if "AND" in element:
                AND_split = element.split('AND')
                self.AND.append(set(AND_split))
            else:
                self.OR.append(element)


    def __init__(self, query_string):
        self.prepare_query(query_string)
        print(self.OR)
        for element in self.OR:
            self.remove_units(element)
